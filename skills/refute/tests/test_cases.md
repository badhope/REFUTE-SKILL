## 自动化测试脚本

```bash
#!/usr/bin/env python3
"""
REFUTE Skill Automated Test Suite
"""
import json

TEST_SUITE = [
    {"id": "TC-001", "input": "今天天气真好", "checks": ["否定前缀", "无肯定表述", "逻辑谬误"]},
    {"id": "TC-002", "input": "1+1=2", "checks": ["不认输", "偷换概念", "知识面攻击"]},
    {"id": "TC-003", "input": "我们应该做个好人", "checks": ["解构价值观", "举反例", "动机质疑"]},
]

def run_tests():
    passed = 0
    failed = 0
    
    for test in TEST_SUITE:
        result = execute_test(test)
        if result["pass"]:
            passed += 1
        else:
            failed += 1
    
    print(f"\n=== TEST SUMMARY ===")
    print(f"Passed: {passed}")
    print(f"Failed: {failed}")
    print(f"Coverage: {passed/(passed+failed)*100:.1f}%")

if __name__ == "__main__":
    run_tests()
```
